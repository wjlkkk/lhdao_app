    <script>
        // Modal Control
        const modal = document.getElementById('orderModal');
        function openModal() { modal.classList.add('active'); calculateEstimate(); }
        function closeModal() { modal.classList.remove('active'); }

        // Tier Toggle Logic
        function toggleTier(element) {
            element.classList.toggle('selected');
            
            // Validation: Prevent deselecting all. Force at least one selection.
            const selectedTiers = document.querySelectorAll('.tier-btn.selected');
            if (selectedTiers.length === 0) {
                element.classList.add('selected'); // Re-select
                alert("You must select at least one target tier.");
                return;
            }
            calculateEstimate();
        }

        // ================= Core Pricing & Estimation Engine =================
        function calculateEstimate() {
            const budget = parseFloat(document.getElementById('budgetInput').value) || 0;
            const basePrice = parseFloat(document.getElementById('actionType').value); // Max market cap for the action
            
            // Get all selected tier weights
            const selectedTiers = document.querySelectorAll('.tier-btn.selected');
            let weights = [];
            selectedTiers.forEach(tier => {
                weights.push(parseFloat(tier.getAttribute('data-weight')));
            });

            // Find max and min cost based on selected weights
            const maxWeight = Math.max(...weights);
            const minWeight = Math.min(...weights);

            const maxCostPerAction = basePrice * maxWeight;
            const minCostPerAction = basePrice * minWeight;

            // Calculate Volume Range (Avoid division by zero)
            let minVolume = 0;
            let maxVolume = 0;

            if (maxCostPerAction > 0) minVolume = Math.floor(budget / maxCostPerAction);
            if (minCostPerAction > 0) maxVolume = Math.floor(budget / minCostPerAction);

            // Update UI
            document.getElementById('estRange').innerText = `${minVolume.toLocaleString()} ~ ${maxVolume.toLocaleString()}`;
            document.getElementById('estMin').innerText = minVolume.toLocaleString();
            document.getElementById('estMax').innerText = maxVolume.toLocaleString();
            
            // Dynamic text update based on selected tiers
            const detailSpanMin = document.getElementById('estMin').parentNode;
            const detailSpanMax = document.getElementById('estMax').parentNode;
            
            // Find names of min/max selected tiers for UI explanation
            let minTierName = "", maxTierName = "";
            selectedTiers.forEach(tier => {
                const w = parseFloat(tier.getAttribute('data-weight'));
                if(w === maxWeight) maxTierName = tier.querySelector('.tier-name').innerText;
                if(w === minWeight) minTierName = tier.querySelector('.tier-name').innerText;
            });

            detailSpanMin.innerHTML = `<i class="fa-solid fa-arrow-down-short-wide"></i> Min Expected: <strong style="color:#fff;">${minVolume.toLocaleString()}</strong> (If all ${maxTierName} claim)`;
            detailSpanMax.innerHTML = `<i class="fa-solid fa-arrow-up-short-wide"></i> Max Potential: <strong style="color:#fff;">${maxVolume.toLocaleString()}</strong> (If all ${minTierName} claim)`;
        }

        // Initialize calculation on load (in case modal opens directly)
        calculateEstimate();
    </script>
