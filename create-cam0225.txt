    <script>
        // === Create Modal Logic ===
        const createModal = document.getElementById('createModal');
        function openCreateModal() { createModal.classList.add('active'); calculateEstimate(); }
        function closeCreateModal() { createModal.classList.remove('active'); }
        function toggleTier(element) { 
            element.classList.toggle('selected'); 
            const selectedTiers = document.querySelectorAll('.tier-btn.selected');
            if(selectedTiers.length === 0) {
                element.classList.add('selected');
                alert("You must select at least one target tier.");
                return;
            }
            calculateEstimate(); 
        }
        function calculateEstimate() {
            const budget = parseFloat(document.getElementById('budgetInput').value) || 0;
            const basePrice = parseFloat(document.getElementById('actionType').value); 
            const selectedTiers = document.querySelectorAll('.tier-btn.selected');
            if(selectedTiers.length === 0) return;
            let weights = []; selectedTiers.forEach(t => weights.push(parseFloat(t.getAttribute('data-weight'))));
            const maxCost = basePrice * Math.max(...weights);
            const minCost = basePrice * Math.min(...weights);
            const minVol = maxCost > 0 ? Math.floor(budget / maxCost) : 0;
            const maxVol = minCost > 0 ? Math.floor(budget / minCost) : 0;
            document.getElementById('estRange').innerText = `${minVol.toLocaleString()} ~ ${maxVol.toLocaleString()}`;
            document.getElementById('estMin').innerText = minVol.toLocaleString();
            document.getElementById('estMax').innerText = maxVol.toLocaleString();
        }

        // === Chart Modal Logic ===
        const chartModal = document.getElementById('chartModal');
        const chartContainer = document.getElementById('chartContainer');

        function openChartModal(title, totalEngagements) {
            document.getElementById('chartModalTitle').innerText = title;
            document.getElementById('chartModalTotal').innerText = totalEngagements;
            chartModal.classList.add('active');
            setTimeout(() => renderChart(totalEngagements), 100);
        }

        function closeChartModal() {
            chartModal.classList.remove('active');
            chartContainer.innerHTML = '<div class="chart-watermark"><i class="fa-solid fa-asterisk"></i> LIGHT HOUSE</div>';
        }

        // 核心修复：使用极其稳定且支持高并发的 DiceBear 像素风头像接口
        function renderChart(targetTotal) {
            const pointsCount = 35; 
            const dataPoints = [];
            for(let i=0; i<pointsCount; i++) {
                const x = (i / (pointsCount - 1)) * 10 - 5;
                const sigmoid = 1 / (1 + Math.exp(-x)); 
                const noise = (Math.random() - 0.5) * 0.03; 
                let val = (sigmoid + noise) * targetTotal;
                if(i > 0 && val < dataPoints[i-1].val) val = dataPoints[i-1].val;
                
                // 更换为 Web3 像素风头像，彻底解决黑点加载不出的问题
                const uniqueAvatarUrl = `https://api.dicebear.com/9.x/pixel-art/svg?seed=user_${i}_${Date.now()}`;

                dataPoints.push({
                    index: i,
                    val: Math.min(val, targetTotal),
                    avatar: uniqueAvatarUrl
                });
            }

            const chartWidth = chartContainer.clientWidth;
            const chartHeight = chartContainer.clientHeight;
            const paddingY = 40; 
            const usableHeight = chartHeight - (paddingY * 2);
            const maxVal = targetTotal * 1.1; 
            
            let svgHtml = `<svg class="chart-svg" viewBox="0 0 ${chartWidth} ${chartHeight}">`;
            for(let j=0; j<=4; j++) {
                const yPos = paddingY + (usableHeight / 4) * j;
                const labelVal = Math.round(maxVal - (maxVal / 4) * j);
                svgHtml += `<line x1="0" y1="${yPos}" x2="${chartWidth}" y2="${yPos}" class="grid-line" />`;
                svgHtml += `<text x="5" y="${yPos - 5}" class="grid-label">${labelVal}</text>`;
            }

            let pathD = "";
            let nodeHtml = "";

            dataPoints.forEach((p, i) => {
                const cx = (i / (pointsCount - 1)) * chartWidth;
                const cy = chartHeight - paddingY - (p.val / maxVal) * usableHeight;
                
                if(i === 0) { pathD += `M ${cx} ${cy} `; } else { pathD += `L ${cx} ${cy} `; }

                if (i > 4 && i % 3 === 0) {
                    const delay = (i * 0.04).toFixed(2);
                    nodeHtml += `<img src="${p.avatar}" class="chart-node" style="left: ${cx}px; top: ${cy}px; animation-delay: ${delay}s;" title="Virtual User ${i}">`;
                }
            });

            svgHtml += `<path class="chart-path" d="${pathD}" />`;
            svgHtml += `</svg>`;

            chartContainer.insertAdjacentHTML('beforeend', svgHtml);
            chartContainer.insertAdjacentHTML('beforeend', nodeHtml);
        }
    </script>
