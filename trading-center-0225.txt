            row.classList.toggle('active');
        }

        // 记录推特跳转 (激活 Step 2)
        function markExecuted(event, verifyBtnId) {
            event.stopPropagation(); 
            const verifyBtn = document.getElementById(verifyBtnId);
            verifyBtn.classList.add('active-step');
        }

        // ★ 核心缝合：验证状态切换 + LUX 金币升腾 + 科技荧光涟漪
        function verifyTask(event, btn, rowId, amount) {
            event.stopPropagation(); 
            if(!btn.classList.contains('active-step') || btn.classList.contains('success')) return;

            // --- 阶段一：进入 Loading 状态 ---
            btn.innerHTML = '';
            btn.classList.remove('active-step');
            btn.classList.add('loading');

            // 模拟 1.5 秒接口校验
            setTimeout(() => {
                
                // --- 阶段二：校验成功，按钮状态变身 ---
                btn.classList.remove('loading');
                btn.classList.add('success');
                btn.innerHTML = `<i class="fa-solid fa-circle-check"></i> +${amount.toFixed(2)} LUX Claimed`;
                
                // --- 动效 1：科技荧光涟漪 (Ripple) ---
                const ripple = document.createElement('div');
                ripple.classList.add('ripple');
                btn.appendChild(ripple);
                setTimeout(() => ripple.remove(), 800); // 动效播完清理DOM

                // --- 动效 2：LUX 金币飘升 (Coin Float) ---
                // 获取按钮在页面中的精确绝对位置
                const rect = btn.getBoundingClientRect();
                const scrollX = window.scrollX || document.documentElement.scrollLeft;
                const scrollY = window.scrollY || document.documentElement.scrollTop;

                // 连续生成 5 个金币
                for(let i=0; i<5; i++) {
                    setTimeout(() => {
                        const coin = document.createElement('i');
                        coin.classList.add('fa-regular', 'fa-sun', 'lux-coin');
                        // 随机散布在按钮宽度范围内，定位到 body 防止被 overflow 遮挡
                        coin.style.left = (rect.left + scrollX + (Math.random() * rect.width * 0.8)) + 'px';
                        coin.style.top = (rect.top + scrollY) + 'px';
                        
                        document.body.appendChild(coin);
                        
                        setTimeout(() => coin.remove(), 1000);
                    }, i * 150); // 错开飘起的时间
                }
                
                // --- 阶段三：3秒后自动折叠完成任务，流程闭环 ---
                setTimeout(() => {
                    document.getElementById(rowId).classList.remove('active');
                }, 3000);

            }, 1500);
        }
    </script>
